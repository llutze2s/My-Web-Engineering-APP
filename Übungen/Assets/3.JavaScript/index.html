        function revocable(fun) {
            var error = false;
            return {
                invoke: function () {
                    if(error){
                        return "Fehler !";
                    } else {
                        return fun.apply(this, arguments);
                    }
                },
                revoke: function () {
                    error = true;
                    fun = null;
                }
            }
        }
